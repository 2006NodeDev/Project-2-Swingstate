set schema 'state-services'; --or whatever the state service schema is called
create table poll_table(
	poll_id serial primary key,
	poll_name text,
	poll_date timestamp,
	democratic_percent int,
	republican_percent int,
	state_id int,
	margin int
);

Cecilia 7/27/20. SQL for states in state service

create table states (
	"state_id" serial primary key,
	"state_name" text not null unique,
	"democrat_candidate" text not null,
	"republican_candidate" text not null,
	"registration_link" text not null,
	"voting_location" text,
	"picture_path" text
);
